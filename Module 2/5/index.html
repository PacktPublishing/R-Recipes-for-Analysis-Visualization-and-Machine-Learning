##   the package animation
library(animation)
library(plot3D)
for (i in 1:100) {
    x = y = seq(0, 2 * pi, length.out = 100)
    z = mesh(x, y)
    u = z$x
    v = z$y
    m = (sin(u) * sin(2 * v)/2)
    n = (sin(2 * u) * cos(v) * cos(v))
    o = (cos(2 * u) * cos(v) * cos(v))
    surf3D(m, n, o, colvar = o, border = "black", colkey = FALSE, 
        theta = i, box = TRUE)
}
